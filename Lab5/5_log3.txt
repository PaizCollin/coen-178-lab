SQL> Create or Replace Procedure setEmpSalary(p_name in VARCHAR, low in INTEGER, high in INTEGER)
  2  As
  3  	     /* Define the local variables you need */
  4  	 l_salary AlphaCoEmp.salary%type;
  5  
  6  BEGIN
  7  	     /* since name is a primary key, set the salary
  8  	     Of the employee where name = p_name.
  9  
 10  	     With an update statement, set the salary of that employee
 11  	     With a randomly generated value between the low and high passed
 12  	     In as parameters
 13  	     */
 14  	 l_salary := ROUND(dbms_random.value(low,high));
 15  	 update AlphaCoEmp
 16  	     set salary = l_salary
 17  	     where name = p_name;
 18  	     commit;
